WordPress React Router 7 Integration Example

This example demonstrates how to integrate the React Router 7 meetings application into WordPress.

Note: This demo loads React from CDN. If you see CORS errors, React might not load properly in development. The WordPress integration works correctly when React is provided by WordPress itself.

WordPress Integration Code

// In your WordPress template or plugin:

// 1. First, enqueue React and ReactDOM (if not already available)
wp_enqueue_script('react', 'https://unpkg.com/react@18/umd/react.production.min.js', array(), '18', false);
wp_enqueue_script('react-dom', 'https://unpkg.com/react-dom@18/umd/react-dom.production.min.js', array('react'), '18', false);

// 2. Enqueue your built WordPress entry file
wp_enqueue_script('meetings-app', get_template_directory_uri() . '/dist/wordpress-entry.js', array('react', 'react-dom'), '1.0', true);

// 3. Add a container div in your template
echo '<div id="react-root"></div>';

// 4. Initialize the React application
wp_add_inline_script('meetings-app', '
    const wordpressModule = window.WordPressMeetings;
    if (wordpressModule && wordpressModule.createMeetingsRouter) {
        const router = wordpressModule.createMeetingsRouter("/meetings");
        const element = React.createElement(wordpressModule.RouterProvider, { router });
        const root = ReactDOM.createRoot(document.getElementById("react-root"));
        root.render(element);
    } else {
        console.error("WordPress meetings module not loaded properly");
    }
');
